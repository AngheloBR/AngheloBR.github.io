02 Variables y tipos de datos

Las variables son contenedores que guardan datos. En Python no necesitas declarar el tipo — Python lo detecta automáticamente.

# Enteros (int)
edad = 20
año = 2026

# Decimales (float)
altura = 1.75
precio = 9.99

# Texto (str)
nombre = "Jaren"
ciudad = 'Lima'

# Booleanos (bool)
activo = True
offline = False

# None (ausencia de valor)
resultado = None

# Verificar el tipo de una variable
print(type(edad))    # <class 'int'>
print(type(nombre))  # <class 'str'>
print(type(activo))  # <class 'bool'>
✓

Los nombres de variables deben ser descriptivos. Usa snake_case: nombre_completo, no NombreCompleto ni nombreCompleto.

03 Operaciones y strings

# Operaciones matemáticas
suma       = 10 + 5    # 15
resta      = 10 - 5    # 5
mult       = 10 * 5    # 50
division   = 10 / 3    # 3.333...
div_entera = 10 // 3   # 3 (descarta decimales)
modulo     = 10 % 3    # 1 (resto de la división)
potencia   = 2 ** 8    # 256

# Operaciones con strings
nombre   = "Jaren"
apellido = "Bailon"
completo = nombre + " " + apellido  # "Jaren Bailon"
repite   = "ha" * 3                 # "hahaha"

# Métodos de string
texto = "  hola mundo  "
print(texto.upper())    # "  HOLA MUNDO  "
print(texto.strip())    # "hola mundo" (sin espacios)
print(texto.split())    # ['hola', 'mundo'] (lista de palabras)
print(len("python"))    # 6 (longitud)

04 Entrada del usuario

input() pausa el programa y espera que el usuario escriba algo. Siempre devuelve un str.

# Leer texto
nombre = input("¿Cómo te llamas? ")
print(f"Hola, {nombre}!")

# Convertir a número (int o float)
edad = int(input("¿Cuántos años tienes? "))
print(f"En 10 años tendrás {edad + 10} años.")

altura = float(input("¿Cuánto mides (en metros)? "))
print(f"Tu altura es {altura:.2f} m.")
⚠

Si el usuario escribe texto cuando esperas un número, int() lanzará un error. Más adelante aprenderás a manejar esto con try/except.

05 Condicionales if / elif / else

Permiten que el programa tome decisiones dependiendo de condiciones.

nota = int(input("Ingresa tu nota (0-100): "))

if nota >= 90:
    print("Excelente — A")
elif nota >= 80:
    print("Bien — B")
elif nota >= 70:
    print("Regular — C")
elif nota >= 60:
    print("Suficiente — D")
else:
    print("Reprobado — F")

Operadores de comparación

x = 10
print(x == 10)   # True  (igual a)
print(x != 5)    # True  (diferente de)
print(x > 8)     # True  (mayor que)
print(x < 5)     # False (menor que)
print(x >= 10)   # True  (mayor o igual)
print(x <= 9)    # False (menor o igual)

# Operadores lógicos
edad = 20
tiene_dni = True
puede_votar = edad >= 18 and tiene_dni  # True
es_menor = not puede_votar              # False

06 Bucles for y while

Bucle for

# Iterar sobre un rango de números
for i in range(5):
    print(i)  # Imprime: 0 1 2 3 4

# range(inicio, fin, paso)
for i in range(1, 11, 2):
    print(i)  # Imprime: 1 3 5 7 9

# Iterar sobre una lista
frutas = ["manzana", "banana", "naranja"]
for fruta in frutas:
    print(f"Fruta: {fruta}")

# enumerate() da índice + valor
for i, fruta in enumerate(frutas):
    print(f"{i}: {fruta}")

Bucle while

# Se repite mientras la condición sea True
contador = 0
while contador < 5:
    print(contador)
    contador += 1  # Equivale a: contador = contador + 1

# break: salir del bucle
while True:
    resp = input("Escribe 'salir' para terminar: ")
    if resp == "salir":
        break

# continue: saltar a la siguiente iteración
for i in range(10):
    if i % 2 == 0:
        continue  # Salta los números pares
    print(i)  # Solo imprime impares: 1 3 5 7 9

07 Funciones

Las funciones agrupan código reutilizable bajo un nombre. Se definen con def y pueden devolver valores con return.

# Función básica
def saludar(nombre):
    return f"Hola, {nombre}!"

mensaje = saludar("Jaren")
print(mensaje)  # "Hola, Jaren!"

# Función con valor por defecto
def potencia(base, exponente=2):
    return base ** exponente

print(potencia(3))     # 9  (exponente=2 por defecto)
print(potencia(3, 3))  # 27

# Función con múltiples retornos (devuelve tupla)
def min_max(lista):
    return min(lista), max(lista)

minimo, maximo = min_max([4, 1, 8, 3, 7])
print(f"Min: {minimo}, Max: {maximo}")  # Min: 1, Max: 8
✓

Si una función es larga, considera dividirla en funciones más pequeñas. Una función debe hacer una sola cosa y hacerla bien.

08 Listas

Las listas son colecciones ordenadas de elementos que pueden modificarse.

# Crear lista
colores = ["rojo", "verde", "azul"]
numeros = [1, 2, 3, 4, 5]
mixta   = [1, "dos", True, 3.14]  # Permite tipos mixtos

# Acceder por índice (empieza en 0)
print(colores[0])   # "rojo"
print(colores[-1])  # "azul" (último elemento)

# Slice (porción)
print(numeros[1:4])  # [2, 3, 4]
print(numeros[:3])   # [1, 2, 3]
print(numeros[2:])   # [3, 4, 5]

# Modificar
colores[0] = "amarillo"

# Métodos principales
colores.append("negro")      # Agrega al final
colores.insert(1, "blanco")  # Inserta en posición
colores.remove("verde")      # Elimina por valor
ultimo = colores.pop()       # Elimina y devuelve el último

print(len(colores))          # Longitud de la lista
colores.sort()               # Ordenar (modifica la lista)
print(sorted(colores))       # Ordenar (devuelve nueva lista)

# Recorrer
for color in colores:
    print(color)

# List comprehension (forma concisa)
cuadrados = [x**2 for x in range(10)]
print(cuadrados)  # [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

09 Diccionarios

Los diccionarios almacenan pares clave: valor. Son perfectos para representar objetos con atributos.

# Crear diccionario
persona = {
    "nombre": "Jaren",
    "edad": 20,
    "ciudad": "Lima",
    "activo": True
}

# Acceder por clave
print(persona["nombre"])            # "Jaren"
print(persona.get("email", "N/A"))  # "N/A" (valor por defecto si no existe)

# Modificar / agregar
persona["edad"] = 21
persona["email"] = "jaren@example.com"

# Eliminar
del persona["ciudad"]
email = persona.pop("email")  # Elimina y devuelve el valor

# Verificar si existe una clave
if "nombre" in persona:
    print("Tiene nombre")

# Iterar
for clave, valor in persona.items():
    print(f"  {clave}: {valor}")

# Solo claves o valores
print(list(persona.keys()))    # ['nombre', 'edad', 'activo']
print(list(persona.values()))  # ['Jaren', 21, True]

# Diccionario de diccionarios
usuarios = {
    "u01": {"nombre": "Ana", "rol": "admin"},
    "u02": {"nombre": "Bob", "rol": "user"},
}
print(usuarios["u01"]["nombre"])  # "Ana"

Ejercicio 1 Calculadora simple

Crea una calculadora que pida dos números y una operación (+, -, *, /), luego muestre el resultado. Maneja la división por cero.

Ver solución 
def calculadora():
    print("=== Calculadora ===")
    a = float(input("Primer número: "))
    op = input("Operación (+  -  *  /): ")
    b = float(input("Segundo número: "))

    if op == "+":
        resultado = a + b
    elif op == "-":
        resultado = a - b
    elif op == "*":
        resultado = a * b
    elif op == "/":
        if b == 0:
            print("Error: no se puede dividir por cero.")
            return
        resultado = a / b
    else:
        print("Operación no válida.")
        return

    print(f"Resultado: {a} {op} {b} = {resultado}")

Ejercicio 2 Adivina el número

El programa elige un número aleatorio entre 1 y 100. El usuario debe adivinarlo. El programa da pistas ("muy alto", "muy bajo") y cuenta los intentos.

Ver solución 
import random

def adivina_el_numero():
    numero = random.randint(1, 100)
    intentos = 0

    print("Adivina el número (entre 1 y 100)")

    while True:
        try:
            intento = int(input("Tu intento: "))
        except ValueError:
            print("Ingresa un número válido.")
            continue

        intentos += 1

        if intento < numero:
            print("Muy bajo ↑")
        elif intento > numero:
            print("Muy alto ↓")
        else:
            print(f"¡Correcto! El número era {numero}.")
            print(f"Lo lograste en {intentos} intento(s).")
            break

Ejercicio 4 Contador de vocales

Escribe una función que reciba un texto y devuelva cuántas vocales contiene. Considera vocales con tilde (á, é, í, ó, ú).

Ver solución 
def contar_vocales(texto):
    vocales = set("aeiouáéíóúAEIOUÁÉÍÓÚ")
    return sum(1 for letra in texto if letra in vocales)

frase = input("Ingresa una frase: ")
total = contar_vocales(frase)
print(f'La frase "{frase}" tiene {total} vocal(es).')

Ejercicio 5 Lista de tareas

Crea un programa de lista de tareas con menú. Debe permitir: ver tareas, agregar tarea, marcar como completada y eliminar tarea.

Ver solución 
def lista_de_tareas():
    tareas = []

    while True:
        print("\n=== Lista de Tareas ===")
        print("1. Ver tareas")
        print("2. Agregar tarea")
        print("3. Completar tarea")
        print("4. Eliminar tarea")
        print("5. Salir")

        opcion = input("\nOpción: ").strip()

        if opcion == "1":
            if not tareas:
                print("No hay tareas.")
            else:
                for i, tarea in enumerate(tareas, 1):
                    estado = "✓" if tarea["hecha"] else "·"
                    print(f"  {estado} {i}. {tarea['nombre']}")

        elif opcion == "2":
            nombre = input("Nueva tarea: ").strip()
            if nombre:
                tareas.append({"nombre": nombre, "hecha": False})
                print(f"  Tarea '{nombre}' agregada.")

        elif opcion == "3":
            if not tareas:
                print("No hay tareas.")
            else:
                for i, t in enumerate(tareas, 1):
                    print(f"  {i}. {t['nombre']}")
                try:
                    idx = int(input("Número de tarea: ")) - 1
                    tareas[idx]["hecha"] = True
                    print("  Tarea completada.")
                except (ValueError, IndexError):
                    print("  Número inválido.")

        elif opcion == "4":
            if not tareas:
                print("No hay tareas.")
            else:
                for i, t in enumerate(tareas, 1):
                    print(f"  {i}. {t['nombre']}")
                try:
                    idx = int(input("Número de tarea a eliminar: ")) - 1
                    eliminada = tareas.pop(idx)
                    print(f"  '{eliminada['nombre']}' eliminada.")
                except (ValueError, IndexError):
                    print("  Número inválido.")

        elif opcion == "5":
            print("¡Hasta luego!")
            break
        else:
            print("Opción no válida.")

lista_de_tareas()
